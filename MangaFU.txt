    <script>
        // Состояние приложения
        let currentUser = null;
        let currentRole = 'guest';
        let cart = [];
        let orders = [];
        let isDarkTheme = false;
        
        // Элементы DOM
        const authForm = document.getElementById('auth-form');
        const authToggle = document.getElementById('auth-toggle');
        const logoutBtn = document.getElementById('logout-btn');
        const currentUserSpan = document.getElementById('current-user');
        const userRoleSpan = document.getElementById('user-role');
        const cartLink = document.getElementById('cart-link');
        const ordersLink = document.getElementById('orders-link');
        const adminLink = document.getElementById('admin-link');
        const userOrders = document.getElementById('user-orders');
        const adminPanel = document.getElementById('admin-panel');
        const cartItems = document.getElementById('cart-items');
        const cartActions = document.getElementById('cart-actions');
        const messageDiv = document.getElementById('message');
        const adminMessageDiv = document.getElementById('admin-message');
        const ordersList = document.getElementById('orders-list');
        
        // Инициализация при загрузке страницы
        document.addEventListener('DOMContentLoaded', function() {
            updateUI();
        });
        
        // Переключение темы
        function toggleTheme() {
            isDarkTheme = !isDarkTheme;
            document.body.classList.toggle('dark-theme', isDarkTheme);
            document.querySelector('.theme-switcher').textContent = isDarkTheme ? 'Светлая тема' : 'Темная тема';
        }
        
        // Переключение формы авторизации
        function toggleAuthForm() {
            authForm.classList.toggle('hidden');
        }
        
        // Обработка авторизации
        function handleAuth(event) {
            event.preventDefault();
            
            const username = document.getElementById('username').value;
            const password = document.getElementById('password').value;
            const role = document.getElementById('role').value;
            
            // В реальном приложении здесь был бы запрос к серверу
            currentUser = username;
            currentRole = role;
            
            // Сохраняем в sessionStorage для имитации сессии
            sessionStorage.setItem('mangafu_user', username);
            sessionStorage.setItem('mangafu_role', role);
            
            // Показываем сообщение
            showMessage(`Добро пожаловать, ${username}! Вы вошли как ${getRoleName(role)}.`);
            
            // Обновляем интерфейс
            updateUI();
            
            // Скрываем форму
            authForm.classList.add('hidden');
        }
        
        // Выход из системы
        function logout() {
            currentUser = null;
            currentRole = 'guest';
            
            // Очищаем sessionStorage
            sessionStorage.removeItem('mangafu_user');
            sessionStorage.removeItem('mangafu_role');
            
            // Показываем сообщение
            showMessage('Вы вышли из системы.');
            
            // Обновляем интерфейс
            updateUI();
        }
        
        // Обновление интерфейса в зависимости от роли
        function updateUI() {
            // Проверяем, есть ли сохраненная сессия
            const savedUser = sessionStorage.getItem('mangafu_user');
            const savedRole = sessionStorage.getItem('mangafu_role');
            
            if (savedUser && savedRole) {
                currentUser = savedUser;
                currentRole = savedRole;
            }
            
            // Обновляем отображение информации о пользователе
            if (currentUser) {
                currentUserSpan.textContent = currentUser;
                userRoleSpan.textContent = getRoleName(currentRole);
                userRoleSpan.classList.remove('hidden');
                authToggle.classList.add('hidden');
                logoutBtn.classList.remove('hidden');
            } else {
                currentUserSpan.textContent = 'Гость';
                userRoleSpan.classList.add('hidden');
                authToggle.classList.remove('hidden');
                logoutBtn.classList.add('hidden');
            }
            
            // Обновляем видимость элементов в зависимости от роли
            // Корзина и заказы доступны всем авторизованным пользователям
            cartLink.classList.toggle('hidden', !currentUser);
            
            // Заказы доступны пользователям и администраторам
            const showOrders = currentUser && (currentRole === 'user' || currentRole === 'admin' || currentRole === 'moderator');
            ordersLink.classList.toggle('hidden', !showOrders);
            userOrders.classList.toggle('hidden', !showOrders);
            
            // Админ-панель доступна только администраторам и модераторам
            const showAdminPanel = currentRole === 'admin' || currentRole === 'moderator';
            adminLink.classList.toggle('hidden', !showAdminPanel);
            adminPanel.classList.toggle('hidden', !showAdminPanel);
            
            // Обновляем корзину
            updateCartDisplay();
            
            // Обновляем список заказов
            updateOrdersDisplay();
        }
        
        // Получение названия роли
        function getRoleName(role) {
            switch(role) {
                case 'admin': return 'Администратор';
                case 'moderator': return 'Модератор';
                case 'user': return 'Пользователь';
                default: return 'Гость';
            }
        }
        
        // Добавление товара в корзину
        function addToCart(mangaTitle, quantity) {
            if (!currentUser) {
                showMessage('Для добавления товаров в корзину необходимо авторизоваться.');
                toggleAuthForm();
                return;
            }
            
            // Проверяем, есть ли уже этот товар в корзине
            const existingItem = cart.find(item => item.title === mangaTitle);
            
            if (existingItem) {
                existingItem.quantity += quantity;
            } else {
                cart.push({
                    title: mangaTitle,
                    quantity: quantity,
                    price: 500 // В реальном приложении цена бралась бы из базы данных
                });
            }
            
            showMessage(`"${mangaTitle}" добавлен в корзину.`);
            updateCartDisplay();
        }
        
        // Обновление отображения корзины
        function updateCartDisplay() {
            if (cart.length === 0) {
                cartItems.innerHTML = '<p>Корзина пуста. Добавьте товары из каталога.</p>';
                cartActions.classList.add('hidden');
                return;
            }
            
            let html = '<h3>Товары в корзине:</h3><ul>';
            let total = 0;
            
            cart.forEach(item => {
                const itemTotal = item.price * item.quantity;
                total += itemTotal;
                html += `<li>${item.title} - ${item.quantity} шт. - ${itemTotal} руб.</li>`;
            });
            
            html += `</ul><p><strong>Итого: ${total} руб.</strong></p>`;
            cartItems.innerHTML = html;
            cartActions.classList.remove('hidden');
        }
        
        // Очистка корзины
        function clearCart() {
            cart = [];
            showMessage('Корзина очищена.');
            updateCartDisplay();
        }
        
        // Оформление заказа
        function checkout() {
            if (cart.length === 0) {
                showMessage('Корзина пуста. Добавьте товары перед оформлением заказа.');
                return;
            }
            
            // Создаем заказ
            const orderId = Date.now(); // Простой ID на основе времени
            const orderTotal = cart.reduce((sum, item) => sum + (item.price * item.quantity), 0);
            
            const order = {
                id: orderId,
                date: new Date().toLocaleDateString(),
                items: [...cart],
                total: orderTotal,
                status: 'В обработке'
            };
            
            orders.push(order);
            
            // Очищаем корзину
            cart = [];
            
            // Показываем сообщение
            showMessage(`Заказ #${orderId} успешно оформлен! Сумма: ${orderTotal} руб.`);
            
            // Обновляем отображение
            updateCartDisplay();
            updateOrdersDisplay();
        }
        
        // Обновление отображения заказов
        function updateOrdersDisplay() {
            if (orders.length === 0) {
                ordersList.innerHTML = '<p>У вас пока нет заказов.</p>';
                return;
            }
            
            let html = '<h3>История заказов:</h3>';
            
            orders.forEach(order => {
                html += `
                    <div class="manga-card">
                        <div class="manga-title">Заказ #${order.id}</div>
                        <div>Дата: ${order.date}</div>
                        <div>Статус: ${order.status}</div>
                        <div>Товаров: ${order.items.length}</div>
                        <div>Сумма: ${order.total} руб.</div>
                    </div>
                `;
            });
            
            ordersList.innerHTML = html;
        }
        
        // Функции для админ-панели
        
        // Показать форму создания манги
        function showCreateForm() {

            document.getElementById('delete-form').classList.add('hidden');
        }
        
        // Скрыть форму создания манги

            document.getElementById('create-form').classList.add('hidden');
        }
        
        // Создание новой манги (заглушка для CRUD Create)

            event.preventDefault();
            
            const title = document.getElementById('manga-title').value;
            const author = document.getElementById('manga-author').value;
            const price = document.getElementById('manga-price').value;
            const stock = document.getElementById('manga-stock').value;
            
            // В реальном приложении здесь был бы запрос к серверу
            showAdminMessage(`Манга "${title}" успешно добавлена в каталог.`);
            
            // Очищаем форму
            document.getElementById('create-form').querySelector('form').reset();
            hideCreateForm();
        }
        
        // Показать форму редактирования манги
        function showEditForm() {
            document.getElementById('edit-form').classList.remove('hidden');
            document.getElementById('create-form').classList.add('hidden');
            document.getElementById('delete-form').classList.add('hidden');
        }

        function hideEditForm() {
            document.getElementById('edit-form').classList.add('hidden');
